SQL Query Analysis
Step 1: Verify CBO user exists
SELECT u.id, u.username, u.firstName, u.lastName, d.designation_name
FROM tbl_user u
INNER JOIN tbl_designation d ON u.designation_id = d.id
WHERE d.designation_name = 'Chief Business Officer'
AND u.id = :user_id

Step 2: Fetch partner users created by CBO
SELECT * FROM tbl_partner_users
WHERE createdBy = :cbo_username
AND (status = 'Active' OR status = 1 OR status IS NULL OR status = '')
ORDER BY first_name ASC, last_name ASC
